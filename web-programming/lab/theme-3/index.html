Код приложения (фрагмент script):

(async () => {
    let response = await fetch('https://kodaktor.ru/j/rates');
    data = await response.json();
    data.forEach(price => {
        document.querySelector('tbody').innerHTML += `<tr><td>${price.name}</td><td>${price.sell}</td><td>${price.buy}</td></tr>`;
    });
})();
